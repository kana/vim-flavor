## Uninstallation steps

    rm -rf ~/.vim-flavor
    rm -rf ~/.vim/pack/flavors

    cd $YOUR_REPOSITORY_FOR_DOTFILES
    rm Flavorfile Flavorfile.lock
    git commit -am 'Farewell to vim-flavor'

    gem uninstall vim-flavor
